/* ── Unit conversion ── */
function convertUnit(name, amount) {
  var n = name.toLowerCase();
  var a = parseFloat(amount) || 0;
  var g = null;
  var keys = Object.keys(CONVERSION_TABLE);
  for (var i = 0; i < keys.length; i++) {
    if (n.indexOf(keys[i]) !== -1) { g = CONVERSION_TABLE[keys[i]]; break; }
  }
  if (!g) return '';
  var cups = a / g;
  if (cups < 0.1) return '';
  var cupStr = cups < 0.12 ? '\u00bc' : cups < 0.3 ? '\u00bc' : cups < 0.6 ? '\u00bd' : cups < 0.9 ? '\u00be' : Math.round(cups) + '';
  if (a < 60) {
    var tsp = (a / 5).toFixed(1).replace(/\.0$/,'');
    var tbsp = (a / 15).toFixed(1).replace(/\.0$/,'');
    if (parseFloat(tsp) < 3) return '(' + tsp + ' \u05db\u05e4\u05d9\u05d5\u05ea)';
    return '(' + tbsp + ' \u05db\u05e4\u05d5\u05ea)';
  }
  return '(' + cupStr + ' \u05db\u05d5\u05e1\u05d5\u05ea)';
}
